//Member controller endpoints
//GET /members - Get all members
curl -X GET http://localhost:8080/members

//POST /members - Create a new member
curl -X POST http://localhost:8080/members \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Doe",
    "spouse": "Jane Doe",
    "phoneNumber": "123-456-7890",
    "address": "123 Main St, Anytown, USA",
    "isAdmin": false
  }'

//PUT /members/{id} - Update a member
curl -X PUT http://localhost:8080/members/1 \
  -H "Content-Type: application/json" \
  -d '{
    "name": "John Updated",
    "spouse": "Jane Doe",
    "phoneNumber": "987-654-3210",
    "address": "456 Elm St, Newtown, USA",
    "isAdmin": true
  }'

//DELETE /members/{id} - Delete a member - Run GET again to confirm
curl -X DELETE http://localhost:8080/members/1
